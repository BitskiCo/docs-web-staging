---
title: "Get token owners"
description:
  "Get the owner(s) of an NFT (ERC721 or ERC1155) as specified by a combination
  of its contractAddress and tokenId."
api: "GET https://api.bitski.com/v1/owners"
---

### Query Parameters

<ParamField query="chainId" type="integer" default="1" required>
  Set the network to get data from. For EVM chains, see a list of available
  chains here.
</ParamField>

<ParamField query="contractAddresses" type="array of string">
  The contract address to get NFT owners of.
</ParamField>

<ParamField query="tokenId" type="string">
  The NFT token id, in decimal or hex format, to filter by.
</ParamField>

<ParamField query="fetchNftMetadata" type="boolean">
  If true, will fetch metadata for NFTs. Only used if `nfts=true` is passed in
  as well.
</ParamField>

### Response

<ParamField body="balances" type="array of object" required>

    <Expandable title ="show objects" >

        <ResponseField name="address" type="string" required>
        The address that owns the token.
        </ResponseField>

        <ResponseField name="amoung" type="string" required>
        Number of tokens owner has in their possesion.
        </ResponseField>

        <ResponseField name="value" type="string" >
        Value of token in USD (used with ERC20 tokens).
        </ResponseField>

        <ResponseField name="blockNumber" type="string" >
        Block number when last evaluated.
        </ResponseField>

        <ResponseField name="token" type="object" required>
        Bitski's standard token model
        </ResponseField>

    </Expandable>

</ParamField>

<ParamField body= "error" type="object" required >
  
    <Expandable title ="show objects" >

        <ResponseField name="message" type="string" />

    </Expandable>

</ParamField>
